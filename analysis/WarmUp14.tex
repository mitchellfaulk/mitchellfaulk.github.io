\noindent Suppose that $f$ satisfies $|f(x)| \leqslant x^2$ for each $x$ in some interval of the origin. Show that 
\begin{enumerate}
\item[(a)] $f$ is continuous at $x = 0$
\item[(b)] $f$ is differentiable at $x = 0$
\item[(c)] $f'(0) = 0$.
\end{enumerate}
